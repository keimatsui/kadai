python api.py "https://api.github.com/repos/AFNetworking/AFNetworking/issues?per_page=100" > AFNetworking-openissues.txt
python api.py "https://api.github.com/repos/AFNetworking/AFNetworking/issues?per_page=100&state=closed" > AFNetworking-closedissues.txt
./jq '.created_at' AFNetworking-openissues.txt | awk '{ printf("%s open\n", $0); }' > AFNetworking-issues.tmp
./jq '.created_at,.closed_at' AFNetworking-closedissues.txt | awk '{ if (NR % 2 == 1) printf("%s open\n", $0); else printf("%s close\n", $0); }' >> AFNetworking-issues.tmp
sort AFNetworking-issues.tmp > AFNetworking-issues.txt
awk 'BEGIN { openissues=0; closedissues=0; } $2=="open" { openissues++; } $2=="close" { closedissues++; } { printf("%s,%d,%d\n", $1, openissues, closedissues) }' AFNetworking-issues.txt > AFNetworking-issues.csv
