python api.py "https://api.github.com/repos/jquery/jquery-mobile/issues?per_page=100" > jquery-mobile-openissues.txt
python api.py "https://api.github.com/repos/jquery/jquery-mobile/issues?per_page=100&state=closed" > jquery-mobile-closedissues.txt
./jq '.created_at' jquery-mobile-openissues.txt | awk '{ printf("%s open\n", $0); }' > jquery-mobile-issues.tmp
./jq '.created_at,.closed_at' jquery-mobile-closedissues.txt | awk '{ if (NR % 2 == 1) printf("%s open\n", $0); else printf("%s close\n", $0); }' >> jquery-mobile-issues.tmp
sort jquery-mobile-issues.tmp > jquery-mobile-issues.txt
awk 'BEGIN { openissues=0; closedissues=0; } $2=="open" { openissues++; } $2=="close" { closedissues++; } { printf("%s,%d,%d\n", $1, openissues, closedissues) }' jquery-mobile-issues.txt > jquery-mobile-issues.csv
