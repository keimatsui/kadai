python api.py "https://api.github.com/repos/plataformatec/devise/issues?per_page=100" > devise-openissues.txt
python api.py "https://api.github.com/repos/plataformatec/devise/issues?per_page=100&state=closed" > devise-closedissues.txt
./jq '.created_at' devise-openissues.txt | awk '{ printf("%s open\n", $0); }' > devise-issues.tmp
./jq '.created_at,.closed_at' devise-closedissues.txt | awk '{ if (NR % 2 == 1) printf("%s open\n", $0); else printf("%s close\n", $0); }' >> devise-issues.tmp
sort devise-issues.tmp > devise-issues.txt
awk 'BEGIN { openissues=0; closedissues=0; } $2=="open" { openissues++; } $2=="close" { closedissues++; } { printf("%s,%d,%d\n", $1, openissues, closedissues) }' devise-issues.txt > devise-issues.csv
bash getIssues.sh plataformatec devise
bash issuesCountChart.sh plataformatec devise
