python api.py "https://api.github.com/repos/octocat/Spoon-Knife/issues?per_page=100" > Spoon-Knife-openissues.txt
python api.py "https://api.github.com/repos/octocat/Spoon-Knife/issues?per_page=100&state=closed" > Spoon-Knife-closedissues.txt
./jq '.created_at' Spoon-Knife-openissues.txt | awk '{ printf("%s open\n", $0); }' > Spoon-Knife-issues.tmp
./jq '.created_at,.closed_at' Spoon-Knife-closedissues.txt | awk '{ if (NR % 2 == 1) printf("%s open\n", $0); else printf("%s close\n", $0); }' >> Spoon-Knife-issues.tmp
sort Spoon-Knife-issues.tmp > Spoon-Knife-issues.txt
awk 'BEGIN { openissues=0; closedissues=0; } $2=="open" { openissues++; } $2=="close" { closedissues++; } { printf("%s,%d,%d\n", $1, openissues, closedissues) }' Spoon-Knife-issues.txt > Spoon-Knife-issues.csv
bash getIssues.sh octocat Spoon-Knife
bash issuesCountChart.sh octocat Spoon-Knife
