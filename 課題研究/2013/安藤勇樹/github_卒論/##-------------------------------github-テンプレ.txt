python api.py "https://api.github.com/repos/URL/issues?per_page=100" > Url-openissues.txt
python api.py "https://api.github.com/repos/URL/issues?per_page=100&state=closed" > Url-closedissues.txt
./jq '.created_at' Url-openissues.txt | awk '{ printf("%s open\n", $0); }' > Url-issues.tmp
./jq '.created_at,.closed_at' Url-closedissues.txt | awk '{ if (NR % 2 == 1) printf("%s open\n", $0); else printf("%s close\n", $0); }' >> Url-issues.tmp
sort Url-issues.tmp > Url-issues.txt
awk 'BEGIN { openissues=0; closedissues=0; } $2=="open" { openissues++; } $2=="close" { closedissues++; } { printf("%s,%d,%d\n", $1, openissues, closedissues) }' Url-issues.txt > Url-issues.csv
bash getIssues.sh shoyuu Url
bash issuesCountChart.sh shoyuu Url
