python api.py "https://api.github.com/repos/github/gitignore/issues?per_page=100" > gitignore-openissues.txt
python api.py "https://api.github.com/repos/github/gitignore/issues?per_page=100&state=closed" > gitignore-closedissues.txt
./jq '.created_at' gitignore-openissues.txt | awk '{ printf("%s open\n", $0); }' > gitignore-issues.tmp
./jq '.created_at,.closed_at' gitignore-closedissues.txt | awk '{ if (NR % 2 == 1) printf("%s open\n", $0); else printf("%s close\n", $0); }' >> gitignore-issues.tmp
sort gitignore-issues.tmp > gitignore-issues.txt
awk 'BEGIN { openissues=0; closedissues=0; } $2=="open" { openissues++; } $2=="close" { closedissues++; } { printf("%s,%d,%d\n", $1, openissues, closedissues) }' gitignore-issues.txt > gitignore-issues.csv
