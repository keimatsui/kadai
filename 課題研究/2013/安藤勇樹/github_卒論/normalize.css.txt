python api.py "https://api.github.com/repos/necolas/normalize.css/issues?per_page=100" > normalize.css-openissues.txt
python api.py "https://api.github.com/repos/necolas/normalize.css/issues?per_page=100&state=closed" > normalize.css-closedissues.txt
./jq '.created_at' normalize.css-openissues.txt | awk '{ printf("%s open\n", $0); }' > normalize.css-issues.tmp
./jq '.created_at,.closed_at' normalize.css-closedissues.txt | awk '{ if (NR % 2 == 1) printf("%s open\n", $0); else printf("%s close\n", $0); }' >> normalize.css-issues.tmp
sort normalize.css-issues.tmp > normalize.css-issues.txt
awk 'BEGIN { openissues=0; closedissues=0; } $2=="open" { openissues++; } $2=="close" { closedissues++; } { printf("%s,%d,%d\n", $1, openissues, closedissues) }' normalize.css-issues.txt > normalize.css-issues.csv
